# react-manage-system
react后台管理系统（双端）
## 安装

```js
{
      "id":"0",		//菜单唯一标识
      "title":"表单组件",	//菜单title
      "key": "key0",	//菜单key
      "url": null,	//外部url(如果存在，则用iframe渲染)
      "leaf": false,	//是否是叶子节点（最终节点）
      "isAuth": false,	//是否有权限
      "isAuthReadable": false,  //是否仅是读权限
      "children":[]	//子菜单
      "route":''	//前端路由
      "icon":''   //菜单icon
      "isBuilding":true //菜单是否建设中
}
```
